import omni.usd
from pxr import UsdGeom, UsdPhysics

j = UsdPhysics.RevoluteJoint.Define(stage, "/World/Cobot/joint_elbow")
j.CreateBody0Rel().SetTargets(["/World/Cobot/Base"])
j.CreateBody1Rel().SetTargets(["/World/Cobot/Forearm"])
