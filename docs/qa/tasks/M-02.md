## Pre-session setup

```python
import omni.usd
import omni.kit.commands
from pxr import UsdGeom, UsdPhysics, Sdf, Gf
ctx = omni.usd.get_context()
stage = ctx.get_stage()
UsdGeom.Xform.Define(stage, "/World")
UsdPhysics.Scene.Define(stage, "/World/PhysicsScene")

# Import real Franka from cuRobo's bundled URDF
franka_urdf = "/home/anton/miniconda3/envs/isaac_lab_env/lib/python3.11/site-packages/curobo/content/assets/robot/franka_description/franka_panda.urdf"
try:
    omni.kit.commands.execute(
        "URDFParseAndImportFile",
        urdf_path=franka_urdf,
        dest_path="/World/Franka",
    )
except Exception:
    robot = UsdGeom.Xform.Define(stage, "/World/Franka").GetPrim()
    UsdPhysics.ArticulationRootAPI.Apply(robot)

# Mark the panda_hand region as having a high-poly custom collision mesh — the
# documented PhysX segfault path. We can't easily author a 12,000-face mesh in
# pure Python, so we annotate the prim with custom data the agent can read.
hand = stage.GetPrimAtPath("/World/Franka/panda_hand")
if hand and hand.IsValid():
    hand.CreateAttribute("custom:gripper_mesh_face_count", Sdf.ValueTypeNames.Int).Set(12000)
    hand.CreateAttribute("custom:collision_approximation_used", Sdf.ValueTypeNames.Token).Set("convexHull")
    hand.CreateAttribute("custom:phys_collision_failure_marker", Sdf.ValueTypeNames.String).Set(
        "12000-face mesh under convex-hull approximation; PhysX limit ~255 polys"
    )

# Table (primitive)
table = UsdGeom.Cube.Define(stage, "/World/Table").GetPrim()
UsdGeom.Xformable(table).AddTranslateOp().Set(Gf.Vec3d(0.4, 0, -0.05))
UsdGeom.Xformable(table).AddScaleOp().Set(Gf.Vec3d(0.6, 0.6, 0.025))
UsdPhysics.CollisionAPI.Apply(table)

# Cylinder object (primitive)
cylinder = UsdGeom.Cylinder.Define(stage, "/World/Cylinder").GetPrim()
UsdGeom.Xformable(cylinder).AddTranslateOp().Set(Gf.Vec3d(0.4, 0, 0.1))
UsdGeom.Cylinder(cylinder).GetRadiusAttr().Set(0.04)
UsdGeom.Cylinder(cylinder).GetHeightAttr().Set(0.15)
UsdPhysics.RigidBodyAPI.Apply(cylinder)
UsdPhysics.CollisionAPI.Apply(cylinder)

```
