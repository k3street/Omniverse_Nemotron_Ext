```python
import omni.usd
from pxr import UsdGeom, UsdPhysics, Sdf, Gf
ctx = omni.usd.get_context()
stage = ctx.get_stage()
UsdGeom.Xform.Define(stage, "/World")
UsdPhysics.Scene.Define(stage, "/World/PhysicsScene")

# Franka articulation placeholder
robot = UsdGeom.Xform.Define(stage, "/World/Franka").GetPrim()
UsdPhysics.ArticulationRootAPI.Apply(robot)
UsdGeom.Cube.Define(stage, "/World/Franka/base")
hand = UsdGeom.Xform.Define(stage, "/World/Franka/panda_hand").GetPrim()
UsdGeom.Xformable(hand).AddTranslateOp().Set(Gf.Vec3d(0, 0, 0.7))

# Custom gripper mesh — flagged as convex-hull approximation (the documented
# segfault path for high-poly + convex-hull beyond PhysX's ~255-polygon limit).
gripper = UsdGeom.Sphere.Define(stage, "/World/Franka/panda_hand/custom_gripper_mesh").GetPrim()
UsdGeom.Sphere(gripper).GetRadiusAttr().Set(0.04)
UsdPhysics.CollisionAPI.Apply(gripper)
UsdPhysics.MeshCollisionAPI.Apply(gripper)
gripper.CreateAttribute("physics:approximation", Sdf.ValueTypeNames.Token).Set("convexHull")

# Table (primitive)
table = UsdGeom.Cube.Define(stage, "/World/Table").GetPrim()
UsdGeom.Xformable(table).AddTranslateOp().Set(Gf.Vec3d(0.4, 0, -0.05))
UsdGeom.Xformable(table).AddScaleOp().Set(Gf.Vec3d(0.6, 0.6, 0.025))
UsdPhysics.CollisionAPI.Apply(table)

# Cylinder object (primitive)
cylinder = UsdGeom.Cylinder.Define(stage, "/World/Cylinder").GetPrim()
UsdGeom.Xformable(cylinder).AddTranslateOp().Set(Gf.Vec3d(0.4, 0, 0.1))
UsdGeom.Cylinder(cylinder).GetRadiusAttr().Set(0.04)
UsdGeom.Cylinder(cylinder).GetHeightAttr().Set(0.15)
UsdPhysics.RigidBodyAPI.Apply(cylinder)
UsdPhysics.CollisionAPI.Apply(cylinder)
```
