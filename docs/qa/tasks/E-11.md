```python
import omni.usd
from pxr import UsdGeom, UsdPhysics
ctx = omni.usd.get_context()
stage = ctx.get_stage()
UsdGeom.Xform.Define(stage, "/World")
UsdPhysics.Scene.Define(stage, "/World/PhysicsScene")
# G1 hand stub with two fingertip children (for contact-sensor placement)
hand = UsdGeom.Xform.Define(stage, "/World/G1Hand").GetPrim()
UsdPhysics.ArticulationRootAPI.Apply(hand)
UsdPhysics.RigidBodyAPI.Apply(hand)
for name in ("Fingertip_L", "Fingertip_R"):
    ft = UsdGeom.Cube.Define(stage, f"/World/G1Hand/{name}").GetPrim()
    UsdPhysics.RigidBodyAPI.Apply(ft)
    UsdPhysics.CollisionAPI.Apply(ft)
# Cardboard box (single-wall crush ~200 N on edge per prose)
box = UsdGeom.Cube.Define(stage, "/World/Box").GetPrim()
UsdPhysics.RigidBodyAPI.Apply(box)
UsdPhysics.CollisionAPI.Apply(box)
UsdPhysics.MassAPI.Apply(box).CreateMassAttr(0.5)
print("seeded E-11: /World/G1Hand (articulation) + Fingertip_L/R + /World/Box (mass=0.5)")
```
