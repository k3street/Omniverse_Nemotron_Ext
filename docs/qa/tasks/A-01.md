import omni.usd
from pxr import UsdGeom, UsdPhysics
ctx = omni.usd.get_context()
stage = ctx.get_stage()
UsdGeom.Xform.Define(stage, "/World")
UsdPhysics.Scene.Define(stage, "/World/PhysicsScene")
# Floor — so the robot has somewhere to sit
floor = UsdGeom.Cube.Define(stage, "/World/Floor").GetPrim()
from pxr import Gf
UsdGeom.Xformable(floor).AddTranslateOp().Set(Gf.Vec3d(0, 0, -0.05))
UsdGeom.Xformable(floor).AddScaleOp().Set(Gf.Vec3d(2, 2, 0.05))
