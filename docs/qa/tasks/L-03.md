## Pre-session setup

```python
import omni.usd
import omni.kit.commands
from pxr import UsdGeom, UsdPhysics, Gf
ctx = omni.usd.get_context()
stage = ctx.get_stage()
UsdGeom.Xform.Define(stage, "/World")
UsdPhysics.Scene.Define(stage, "/World/PhysicsScene")

# Import real UR10e so the EOAT is at a known robot frame
ur10e_urdf = "/home/anton/miniconda3/envs/isaac_lab_env/lib/python3.11/site-packages/curobo/content/assets/robot/ur_description/ur10e.urdf"
try:
    omni.kit.commands.execute(
        "URDFParseAndImportFile",
        urdf_path=ur10e_urdf,
        dest_path="/World/UR10e",
    )
except Exception:
    robot = UsdGeom.Xform.Define(stage, "/World/UR10e").GetPrim()
    UsdPhysics.ArticulationRootAPI.Apply(robot)

# Fixture — flat plate
fixture = UsdGeom.Cube.Define(stage, "/World/Fixture").GetPrim()
UsdGeom.Xformable(fixture).AddTranslateOp().Set(Gf.Vec3d(0.5, 0, 0.3))
UsdGeom.Xformable(fixture).AddScaleOp().Set(Gf.Vec3d(0.3, 0.2, 0.02))
UsdPhysics.CollisionAPI.Apply(fixture)

# EOAT tip — small sphere just above the fixture top
eoat = UsdGeom.Sphere.Define(stage, "/World/EOAT").GetPrim()
UsdGeom.Xformable(eoat).AddTranslateOp().Set(Gf.Vec3d(0.5, 0, 0.45))
UsdGeom.Sphere(eoat).GetRadiusAttr().Set(0.01)
UsdPhysics.CollisionAPI.Apply(eoat)
# Lisa expects the gap between fixture top (~z=0.32) and EOAT center (z=0.45) ≈ 130 mm
```
